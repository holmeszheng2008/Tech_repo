@Bean
public WebServerFactoryCustomizer<TomcatServletWebServerFactory> tomcatSslStoreCustomizer() {
	String keyStorePassword = responseEntityBody.get("password").toString();
	String p12Base64 = responseEntityBody.get("certificate").toString();
	
	byte[] decodedBytes = Base64.getDecoder().decode(p12Base64);
	InputStream keyStoreFile = new ByteArrayInputStream(decodedBytes);
	KeyStore keyStore;
	
	try(InputStream is = keyStoreFile){
		keyStore = KeyStore.getInstance("PKCS12")
		keyStore.load(is, keyStorePassword.toCharArray());
	} catch (Exception ex){
	}

	return tomcatFactory -> {
		tomcatFactory.getSsl().setKeyPassword(keyStorePassword);
		tomcatFactory.setSslStoreProvider(new SslStoreProvider() {
			@Override
			public KeyStore getKeyStore() { return keyStore;}
			
			@Override
			public KeyStore getTrustStore() { return null;}
		});
	}
}
